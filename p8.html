############################### #8)various morphological image processing techniques import cv2 import numpy as np inp_img = 'images/erod.jpg' image = cv2.imread(inp_img, 0) invert = cv2.bitwise_not(image) kernel = np.ones((5, 5), np.uint8)#structuring element erosion = cv2.erode(invert, kernel, iterations=1) dilation = cv2.dilate(invert, kernel, iterations=1) opening = cv2.morphologyEx(invert, cv2.MORPH_OPEN, kernel) closing = cv2.morphologyEx(invert, cv2.MORPH_CLOSE, kernel) cv2.imshow('Original Image', image) cv2.imshow('Erosion', erosion) cv2.imshow('Dilation', dilation) cv2.imshow('Opening', opening) cv2.imshow('Closing', closing) cv2.waitKey(0) cv2.destroyAllWindows()
